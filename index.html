function doPost(e) {
  try {
    const raw = e.postData ? e.postData.contents : '{}';
    const data = JSON.parse(raw);
    const ss = SpreadsheetApp.getActiveSpreadsheet();

    // Guardar DB (visitas, hoteles, acuerdos, programas)
    if (data.action === 'writeDB') {
      const sheet = ss.getSheetByName('DB') || ss.insertSheet('DB');
      sheet.getRange('A1').setValue(data.data);
      return ok();
    }

    // Eliminar proveedor por RUT empresa
    if (data.action === 'deleteProveedor') {
      const sheet = ss.getSheetByName('Proveedores');
      if (sheet && data.rutEmpresa) {
        const lastRow = sheet.getLastRow();
        if (lastRow > 1) {
          const ruts = sheet.getRange(2, 6, lastRow - 1, 1).getValues();
          for (let i = ruts.length - 1; i >= 0; i--) {
            if (ruts[i][0].toString().trim() === data.rutEmpresa.toString().trim()) {
              sheet.deleteRow(i + 2);
            }
          }
        }
      }
      return ok();
    }

    // Upsert hoja Proveedores — actualiza existentes por RUT empresa, agrega nuevos
    if (data.action === 'writeProveedores') {
      let sheet = ss.getSheetByName('Proveedores');
      if (!sheet) {
        sheet = ss.insertSheet('Proveedores');
        // Hoja nueva — escribir todo directo
        const allRows = [data.headers, ...data.rows];
        sheet.getRange(1, 1, allRows.length, allRows[0].length).setValues(allRows);
        sheet.getRange(1, 1, 1, data.headers.length)
          .setBackground('#00778A').setFontColor('#FFFFFF').setFontWeight('bold');
        return ok();
      }
      // Hoja existente — upsert por RUT empresa (col 6, índice 5)
      const lastRow = sheet.getLastRow();
      const lastCol = Math.max(sheet.getLastColumn(), data.headers.length);
      // Escribir/actualizar encabezados
      sheet.getRange(1, 1, 1, data.headers.length).setValues([data.headers])
        .setBackground('#00778A').setFontColor('#FFFFFF').setFontWeight('bold');
      // Leer RUTs existentes (col F = índice 6 = columna 6)
      const rutCol = 6; // RUT EMPRESA
      const existingRuts = {};
      if (lastRow > 1) {
        const rutValues = sheet.getRange(2, rutCol, lastRow - 1, 1).getValues();
        rutValues.forEach((r, i) => {
          if (r[0]) existingRuts[r[0].toString().trim()] = i + 2; // fila en Sheet (1-based, +1 header)
        });
      }
      // Upsert cada fila
      data.rows.forEach(row => {
        const rut = (row[5] || '').toString().trim(); // índice 5 = RUT EMPRESA
        if (rut && existingRuts[rut]) {
          // Actualizar fila existente
          sheet.getRange(existingRuts[rut], 1, 1, row.length).setValues([row]);
        } else {
          // Nueva fila — agregar al final
          const nextRow = sheet.getLastRow() + 1;
          sheet.getRange(nextRow, 1, 1, row.length).setValues([row]);
          if (rut) existingRuts[rut] = nextRow;
        }
      });
      return ok();
    }

    return ok();
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({ok:false,error:err.message}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName('DB');
  const val = sheet ? sheet.getRange('A1').getValue() : '{}';
  return ContentService
    .createTextOutput(val)
    .setMimeType(ContentService.MimeType.TEXT);
}

function ok() {
  return ContentService
    .createTextOutput(JSON.stringify({ok:true}))
    .setMimeType(ContentService.MimeType.JSON);
}
